```typescript
import type { PositionSizingInput, PositionSizingResult } from './types.js';

/**
 * Calculate position size using fixed percentage of bankroll (D-01, D-04)
 * NOT Kelly Criterion — simpler, predictable, less AI hallucination
 * Default: 5% of bankroll per bet
 * 
 * Formula: positionValue = bankroll * maxPositionPct
 *          positionSize = positionValue / odds
 */
export function calculatePositionSize(input: PositionSizingInput): PositionSizingResult {
  const { bankroll, odds, minPosition, maxPositionPct } = input;
  
  // Step 1: Calculate max position value (5% of bankroll per D-04)
  const maxPositionValue = bankroll * maxPositionPct;
  
  // Step 2: Calculate position size in tokens
  const positionSize = maxPositionValue / odds;
  const positionValue = positionSize * odds;
  
  // Step 3: Enforce minimum threshold (D-03)
  if (positionSize < minPosition) {
    return {
      positionSize: 0,
      positionValue: 0,
      maxAllowed: 0,
      isValid: false,
      reason: `Position size ${positionSize.toFixed(2)} below minimum ${minPosition} tokens — skip bet (D-03)`,
      exposureCheck: input.exposureCaps.find(e => e.category === input.category) || {
        category: input.category,
        totalValue: 0,
        maxExposurePct: maxPositionPct,
        currentPct: 0,
        isWithinLimit: true,
      },
    };
  }
  
  return {
    positionSize,
    positionValue,
    maxAllowed: maxPositionValue,
    isValid: true,
    exposureCheck: input.exposureCaps.find(e => e.category === input.category) || {
      category: input.category,
      totalValue: 0,
      maxExposurePct: maxPositionPct,
      currentPct: 0,
      isWithinLimit: true,
    },
  };
}

/**
 * Get effective position percentage based on research quality (D-07, D-08)
 * Low quality research → conservative (10%)
 * High quality research → moderate (20%)
 */
export function getEffectivePositionPct(
  basePct: number,
  researchQuality: 'low' | 'medium' | 'high'
): number {
  switch (researchQuality) {
    case 'low':
      return Math.min(basePct, 0.10); // Conservative: max 10%
    case 'high':
      return Math.min(basePct * 2, 0.20); // Moderate: up to 20%
    case 'medium':
    default:
      return basePct; // Use configured value (5%)
  }
}
```

```typescript
import type { BankrollState, CategoryExposure, OpenPosition } from './types.js';

/**
 * Default exposure caps per category (D-05)
 * Conservative to moderate range: 10-20% of bankroll
 */
const DEFAULT_EXPOSURE_CAPS: Record<string, number> = {
  crypto: 0.20,      // 20% — high volatility category
  financial: 0.15,   // 15% — moderate
  news: 0.10,        // 10% — high uncertainty
  sports: 0.10,      // 10% — unpredictable
  default: 0.10,     // 10% — conservative default
};

/**
 * Calculate current exposure for a category (D-05)
 * Sum of all open positions in that category
 */
export function calculateCategoryExposure(
  category: string,
  openPositions: OpenPosition[],
  bankroll: number
): CategoryExposure {
  const categoryPositions = openPositions.filter(p => p.category === category);
  const totalValue = categoryPositions.reduce((sum, p) => sum + p.value, 0);
  const maxExposurePct = DEFAULT_EXPOSURE_CAPS[category] || DEFAULT_EXPOSURE_CAPS.default;
  const currentPct = bankroll > 0 ? totalValue / bankroll : 0;
  
  return {
    category,
    totalValue,
    maxExposurePct,
    currentPct,
    isWithinLimit: currentPct <= maxExposurePct,
  };
}

/**
 * Check all category exposures for bankroll state
 */
export function calculateAllExposures(
  bankrollState: BankrollState
): CategoryExposure[] {
  const exposureMap = new Map<string, CategoryExposure>();
  
  // Initialize all known categories
  for (const category of Object.keys(DEFAULT_EXPOSURE_CAPS)) {
    if (category !== 'default') {
      exposureMap.set(category, {
        category,
        totalValue: 0,
        maxExposurePct: DEFAULT_EXPOSURE_CAPS[category],
        currentPct: 0,
        isWithinLimit: true,
      });
    }
  }
  
  // Sum up open positions by category
  for (const position of bankrollState.openPositions) {
    const existing = exposureMap.get(position.category);
    if (existing) {
      existing.totalValue += position.value;
      existing.currentPct = existing.totalValue / bankrollState.totalBankroll;
      existing.isWithinLimit = existing.currentPct <= existing.maxExposurePct;
    }
  }
  
  return Array.from(exposureMap.values());
}

/**
 * Check if adding a new position would exceed category cap (D-05 to D-08)
 * Returns CategoryExposure AFTER the hypothetical new position
 */
export function checkExposureCap(
  category: string,
  newPositionValue: number,
  bankrollState: BankrollState,
  researchQuality: 'low' | 'medium' | 'high'
): CategoryExposure {
  const currentExposure = calculateCategoryExposure(
    category,
    bankrollState.openPositions,
    bankrollState.totalBankroll
  );
  
  // Adjust max based on research quality (D-07, D-08)
  let effectiveMax = currentExposure.maxExposurePct;
  if (researchQuality === 'low') {
    effectiveMax = Math.min(effectiveMax, 0.10); // Conservative
  } else if (researchQuality === 'high') {
    effectiveMax = Math.min(effectiveMax * 1.5, 0.20); // Moderate
  }
  
  const newTotalValue = currentExposure.totalValue + newPositionValue;
  const newPct = newTotalValue / bankrollState.totalBankroll;
  
  return {
    category,
    totalValue: newTotalValue,
    maxExposurePct: effectiveMax,
    currentPct: newPct,
    isWithinLimit: newPct <= effectiveMax,
  };
}

/**
 * Get exposure caps config (read from config.yaml if available)
 */
export function getExposureCapsFromConfig(config: { bankroll?: { categoryCaps?: Record<string, number> } }): Record<string, number> {
  if (config?.bankroll?.categoryCaps) {
    return { ...DEFAULT_EXPOSURE_CAPS, ...config.bankroll.categoryCaps };
  }
  return DEFAULT_EXPOSURE_CAPS;
}
```

Key points:
- 10-20% category cap per D-05, no aggressive 30%+ per D-06
- Research quality adjusts cap per D-07, D-08
- Default caps by category: crypto 20%, financial 15%, news 10%, sports 10%
