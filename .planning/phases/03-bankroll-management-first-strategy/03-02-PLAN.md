```typescript
import type { OrderBook } from '../types/index.js';

/**
 * Slippage protection configuration (D-09)
 * 10% maximum slippage tolerance — professional standard
 * Polymarket is slower than Binance, 10% is professional equilibrium
 */
export interface SlippageConfig {
  maxSlippagePct: number;  // 0.10 = 10%
}

export const DEFAULT_SLIPPAGE_CONFIG: SlippageConfig = {
  maxSlippagePct: 0.10,
};

/**
 * Check if a price move exceeds slippage tolerance (D-10)
 * 
 * @param entryPrice - Price when decision was made
 * @param currentPrice - Current market price
 * @param config - Slippage configuration
 * @returns Object with passed status and details
 */
export function checkSlippage(
  entryPrice: number,
  currentPrice: number,
  config: SlippageConfig = DEFAULT_SLIPPAGE_CONFIG
): { passed: boolean; slippagePct: number; threshold: number; message: string } {
  const priceChange = Math.abs(currentPrice - entryPrice);
  const slippagePct = entryPrice > 0 ? priceChange / entryPrice : 0;
  
  if (slippagePct > config.maxSlippagePct) {
    return {
      passed: false,
      slippagePct,
      threshold: config.maxSlippagePct,
      message: `Slippage ${(slippagePct * 100).toFixed(2)}% exceeds max ${(config.maxSlippagePct * 100).toFixed(0)}% — abort bet (D-10)`,
    };
  }
  
  return {
    passed: true,
    slippagePct,
    threshold: config.maxSlippagePct,
    message: `Slippage ${(slippagePct * 100).toFixed(2)}% within tolerance`,
  };
}

/**
 * Get effective price considering slippage
 * Used for limit order pricing
 */
export function getEffectivePrice(
  midPrice: number,
  side: 'BUY' | 'SELL',
  slippageConfig: SlippageConfig = DEFAULT_SLIPPAGE_CONFIG
): number {
  // For a BUY order, we pay slightly more than mid (worst case)
  // For a SELL order, we receive slightly less than mid (worst case)
  const adjustment = midPrice * slippageConfig.maxSlippagePct;
  
  if (side === 'BUY') {
    return midPrice + adjustment;
  } else {
    return midPrice - adjustment;
  }
}

/**
 * Validate orderbook prices before placing order
 */
export function validateOrderBookPrices(orderbook: OrderBook, side: 'BUY' | 'SELL'): boolean {
  const { bestBid, bestAsk } = getBestPrices(orderbook);
  
  if (side === 'BUY' && bestAsk === null) return false;
  if (side === 'SELL' && bestBid === null) return false;
  
  return true;
}

// Import helper from clob
function getBestPrices(orderbook: OrderBook): { bestBid: number | null; bestAsk: number | null } {
  return {
    bestBid: orderbook.bids[0]?.price || null,
    bestAsk: orderbook.asks[0]?.price || null,
  };
}
```

```typescript
import type { OrderBook } from '../types/index.js';
import { getOrderBook, getMidPrice, hasLiquidity, getBestPrices } from '../api/clob.js';
import { checkSlippage, DEFAULT_SLIPPAGE_CONFIG } from './slippage.js';
import type { SlippageConfig } from './slippage.js';

/**
 * Order type selection (D-12 to D-14)
 * - Limit orders by default (define specific price, wait for market)
 * - Market order as fallback if limit order fails
 * - Hybrid: try limit first, fall back to market if needed
 */
export type OrderType = 'LIMIT' | 'MARKET';

/**
 * Order request parameters
 */
export interface OrderRequest {
  marketId: string;
  tokenId: string;           // CLOB token ID
  side: 'BUY' | 'SELL';
  size: number;              // Number of shares
  limitPrice?: number;       // For LIMIT orders
  slippageConfig?: SlippageConfig;
}

/**
 * Order result
 */
export interface OrderResult {
  success: boolean;
  orderType: OrderType;
  executedPrice: number;
  executedSize: number;
  slippageCheck: { passed: boolean; slippagePct: number; message: string };
  error?: string;
  // Fallback tracking
  triedLimitFirst: boolean;
  fellBackToMarket: boolean;
}

/**
 * Determine order type based on parameters (D-12 to D-14)
 * Default to LIMIT, fallback to MARKET if limit fails or times out
 */
export async function placeOrder(request: OrderRequest): Promise<OrderResult> {
  const slippageConfig = request.slippageConfig || DEFAULT_SLIPPAGE_CONFIG;
  
  // Get current market state
  const orderbook = await getOrderBook(request.tokenId);
  const midPrice = getMidPrice(orderbook);
  
  if (midPrice === null) {
    return {
      success: false,
      orderType: 'MARKET',
      executedPrice: 0,
      executedSize: 0,
      slippageCheck: { passed: false, slippagePct: 0, message: 'No market prices available' },
      error: 'Orderbook empty',
      triedLimitFirst: false,
      fellBackToMarket: false,
    };
  }
  
  // Check slippage before placing order (EXEC-04)
  const entryPrice = request.limitPrice || midPrice;
  const currentBest = request.side === 'BUY' 
    ? orderbook.asks[0]?.price 
    : orderbook.bids[0]?.price;
    
  if (currentBest === undefined) {
    return {
      success: false,
      orderType: 'MARKET',
      executedPrice: 0,
      executedSize: 0,
      slippageCheck: { passed: false, slippagePct: 0, message: 'No liquidity' },
      error: 'No liquidity at any price',
      triedLimitFirst: false,
      fellBackToMarket: false,
    };
  }
  
  const slippageResult = checkSlippage(entryPrice, currentBest, slippageConfig);
  
  if (!slippageResult.passed) {
    // D-10: Abort if price moved beyond threshold
    return {
      success: false,
      orderType: 'LIMIT',
      executedPrice: 0,
      executedSize: 0,
      slippageCheck: slippageResult,
      error: slippageResult.message,
      triedLimitFirst: true,
      fellBackToMarket: false,
    };
  }
  
  // D-12 to D-14: Try limit first, fallback to market
  const limitResult = await tryLimitOrder(request, orderbook, slippageConfig);
  
  if (limitResult.success) {
    return {
      ...limitResult,
      triedLimitFirst: true,
      fellBackToMarket: false,
    };
  }
  
  // Limit failed — fallback to market (D-13)
  console.log(`Limit order failed: ${limitResult.error}, falling back to market order`);
  
  const marketResult = await tryMarketOrder(request, orderbook);
  
  return {
    ...marketResult,
    triedLimitFirst: true,
    fellBackToMarket: true,
  };
}

/**
 * Try placing a limit order at specified or mid price
 */
async function tryLimitOrder(
  request: OrderRequest,
  orderbook: OrderBook,
  slippageConfig: SlippageConfig
): Promise<Omit<OrderResult, 'triedLimitFirst' | 'fellBackToMarket'>> {
  const { bestBid, bestAsk } = getBestPrices(orderbook);
  const limitPrice = request.limitPrice || getMidPrice(orderbook);
  
  if (limitPrice === null) {
    return {
      success: false,
      orderType: 'LIMIT',
      executedPrice: 0,
      executedSize: 0,
      slippageCheck: { passed: false, slippagePct: 0, message: 'No mid price' },
      error: 'Cannot determine limit price',
    };
  }
  
  // Check if limit price is within slippage tolerance
  const currentPrice = request.side === 'BUY' ? bestAsk : bestBid;
  if (currentPrice !== null) {
    const slippageResult = checkSlippage(limitPrice, currentPrice, slippageConfig);
    if (!slippageResult.passed) {
      return {
        success: false,
        orderType: 'LIMIT',
        executedPrice: 0,
        executedSize: 0,
        slippageCheck: slippageResult,
        error: `Limit price ${limitPrice} would exceed slippage: ${slippageResult.message}`,
      };
    }
  }
  
  // For now, simulate limit order (actual implementation would call CLOB API)
  // In production: await clobClient.createLimitOrder({ tokenId, side, size, price: limitPrice })
  return {
    success: true,
    orderType: 'LIMIT',
    executedPrice: limitPrice,
    executedSize: request.size,
    slippageCheck: { passed: true, slippagePct: 0, message: 'Limit order placed' },
  };
}

/**
 * Fallback to market order (D-13)
 */
async function tryMarketOrder(
  request: OrderRequest,
  orderbook: OrderBook
): Promise<Omit<OrderResult, 'triedLimitFirst' | 'fellBackToMarket'>> {
  const { bestBid, bestAsk } = getBestPrices(orderbook);
  const marketPrice = request.side === 'BUY' ? bestAsk : bestBid;
  
  if (marketPrice === null) {
    return {
      success: false,
      orderType: 'MARKET',
      executedPrice: 0,
      executedSize: 0,
      slippageCheck: { passed: false, slippagePct: 0, message: 'No market price' },
      error: 'No market price available for fallback',
    };
  }
  
  // Market order: execute immediately at current best price
  // In production: await clobClient.createMarketOrder({ tokenId, side, size })
  return {
    success: true,
    orderType: 'MARKET',
    executedPrice: marketPrice,
    executedSize: request.size,
    slippageCheck: { passed: true, slippagePct: 0, message: 'Market order executed' },
  };
}
```

```typescript
import type { Market, OrderBook } from '../types/index.js';
import { getOrderBook, getMidPrice, getBestPrices } from '../api/clob.js';

/**
 * Arbitrage detection configuration (D-15 to D-18)
 * 
 * Formula: YES_price + NO_price < $0.99 (accounting for ~1% fees)
 * If arbitrage detected → bet on both YES and NO sides
 * Guaranteed profit when YES + NO < $0.99
 */

export interface ArbitrageConfig {
  maxCombinedPrice: number;  // 0.99 (D-16) — includes ~1% fee buffer
  minProfitPct: number;      // Minimum profit margin to execute
}

export const DEFAULT_ARBITRAGE_CONFIG: ArbitrageConfig = {
  maxCombinedPrice: 0.99,  // D-16
  minProfitPct: 0.005,      // 0.5% minimum profit after fees
};

/**
 * Arbitrage opportunity detected
 */
export interface ArbitrageOpportunity {
  marketId: string;
  yesPrice: number;
  noPrice: number;
  combinedPrice: number;
  feeBuffer: number;         // 1 - combinedPrice (profit margin)
  profitPct: number;         // Fee buffer as percentage
  isViable: boolean;         // Within config limits
  reason: string;
}

/**
 * Arbitrage result after placing both bets
 */
export interface ArbitrageResult {
  success: boolean;
  opportunity: ArbitrageOpportunity;
  yesOrder: {
    side: 'BUY';
    price: number;
    size: number;
    value: number;
  };
  noOrder: {
    side: 'BUY';
    price: number;
    size: number;
    value: number;
  };
  totalCost: number;
  guaranteedProfit: number;  // If both orders fill at quoted prices
}

/**
 * Detect arbitrage opportunity in a market (D-15, D-16)
 * 
 * Binary complement: YES + NO should = $1.00 (par)
 * With ~1% Polymarket fees, YES + NO typically = $0.99
 * If YES + NO < $0.99 → arbitrage exists
 */
export async function detectArbitrage(market: Market): Promise<ArbitrageOpportunity> {
  const [yesTokenId, noTokenId] = market.clobTokenIds;
  
  // Fetch both order books simultaneously
  const [yesOrderBook, noOrderBook] = await Promise.all([
    getOrderBook(yesTokenId),
    getOrderBook(noTokenId),
  ]);
  
  const yesMidPrice = getMidPrice(yesOrderBook);
  const noMidPrice = getMidPrice(noOrderBook);
  
  if (yesMidPrice === null || noMidPrice === null) {
    return {
      marketId: market.id,
      yesPrice: 0,
      noPrice: 0,
      combinedPrice: 0,
      feeBuffer: 0,
      profitPct: 0,
      isViable: false,
      reason: 'Could not determine mid prices for YES and NO',
    };
  }
  
  const combinedPrice = yesMidPrice + noMidPrice;
  const feeBuffer = 1 - combinedPrice; // D-16: 1% fee buffer
  const profitPct = feeBuffer;
  
  // D-16: Check if arbitrage exists (YES + NO < $0.99)
  const isViable = combinedPrice < DEFAULT_ARBITRAGE_CONFIG.maxCombinedPrice;
  
  let reason: string;
  if (isViable) {
    reason = `Arbitrage detected: YES (${yesMidPrice.toFixed(4)}) + NO (${noMidPrice.toFixed(4)}) = ${combinedPrice.toFixed(4)} < ${DEFAULT_ARBITRAGE_CONFIG.maxCombinedPrice} (D-16)`;
  } else {
    reason = `No arbitrage: YES (${yesMidPrice.toFixed(4)}) + NO (${noMidPrice.toFixed(4)}) = ${combinedPrice.toFixed(4)} >= ${DEFAULT_ARBITRAGE_CONFIG.maxCombinedPrice}`;
  }
  
  return {
    marketId: market.id,
    yesPrice: yesMidPrice,
    noPrice: noMidPrice,
    combinedPrice,
    feeBuffer,
    profitPct,
    isViable,
    reason,
  };
}

/**
 * Calculate arbitrage bet sizes for both YES and NO
 * 
 * D-17: If arbitrage detected → bet on both YES and NO sides
 * D-18: Guaranteed profit when YES + NO < $0.99
 * 
 * Strategy: Spend X on YES and (Y = X * odds) on NO such that:
 * - If YES wins: YES_payout - cost = profit
 * - If NO wins: NO_payout - cost = profit
 */
export function calculateArbitrageBetSize(
  opportunity: ArbitrageOpportunity,
  totalStake: number  // Total amount to risk
): { yesSize: number; noSize: number; yesValue: number; noValue: number; guaranteedProfit: number } {
  const { yesPrice, noPrice, combinedPrice } = opportunity;
  
  // Calculate sizes for equal profit regardless of outcome
  // If YES wins: we get (yesSize) and lose (noValue)
  // If NO wins: we get (noSize) and lose (yesValue)
  
  // For guaranteed profit:
  // yesValue / yesPrice = noValue / noPrice (sizes that equalize payouts)
  // yesValue + noValue = totalStake
  
  // Solving:
  // yesValue = totalStake * yesPrice / (yesPrice + noPrice)
  // noValue = totalStake * noPrice / (yesPrice + noPrice)
  
  const yesValue = totalStake * yesPrice / combinedPrice;
  const noValue = totalStake * noPrice / combinedPrice;
  
  const yesSize = yesValue / yesPrice;  // Shares of YES
  const noSize = noValue / noPrice;    // Shares of NO
  
  // Guaranteed profit at current prices
  const yesPayoutIfYesWins = yesSize;  // 1 share = 1 USDC at full payout
  const noPayoutIfNoWins = noSize;     // 1 share = 1 USDC at full payout
  const cost = yesValue + noValue;
  const guaranteedProfit = Math.min(yesPayoutIfYesWins, noPayoutIfNoWins) - cost;
  
  return {
    yesSize,
    noSize,
    yesValue,
    noValue,
    guaranteedProfit,
  };
}

/**
 * Execute arbitrage opportunity (D-17: bet both YES and NO)
 * 
 * This is the ONLY strategy that survived live testing per the context.
 * The bot places both YES and NO bets simultaneously.
 */
export async function executeArbitrage(
  market: Market,
  totalStake: number,
  config: ArbitrageConfig = DEFAULT_ARBITRAGE_CONFIG
): Promise<ArbitrageResult | null> {
  // First detect arbitrage
  const opportunity = await detectArbitrage(market);
  
  if (!opportunity.isViable) {
    console.log(`No arbitrage in ${market.question}: ${opportunity.reason}`);
    return null;
  }
  
  console.log(`Arbitrage found in ${market.question}: ${opportunity.reason}`);
  
  // Calculate bet sizes
  const bets = calculateArbitrageBetSize(opportunity, totalStake);
  
  if (bets.guaranteedProfit <= 0) {
    console.log(`Arbitrage not profitable after calculation: profit=${bets.guaranteedProfit.toFixed(4)}`);
    return null;
  }
  
  // Log the opportunity
  console.log(`
========== ARBITRAGE OPPORTUNITY ==========
Market: ${market.question}
YES: ${opportunity.yesPrice.toFixed(4)} × ${bets.yesSize.toFixed(2)} = $${bets.yesValue.toFixed(2)}
NO:  ${opportunity.noPrice.toFixed(4)} × ${bets.noSize.toFixed(2)} = $${bets.noValue.toFixed(2)}
Combined: $${(bets.yesValue + bets.noValue).toFixed(2)}
Guaranteed Profit: $${bets.guaranteedProfit.toFixed(2)}
=========================================
  `);
  
  return {
    success: true,
    opportunity,
    yesOrder: {
      side: 'BUY',
      price: opportunity.yesPrice,
      size: bets.yesSize,
      value: bets.yesValue,
    },
    noOrder: {
      side: 'BUY',
      price: opportunity.noPrice,
      size: bets.noSize,
      value: bets.noValue,
    },
    totalCost: bets.yesValue + bets.noValue,
    guaranteedProfit: bets.guaranteedProfit,
  };
}

/**
 * Quick arbitrage check using best bid/ask instead of full mid price
 * Useful for fast screening of many markets
 */
export async function quickArbitrageCheck(market: Market): Promise<boolean> {
  const [yesTokenId, noTokenId] = market.clobTokenIds;
  
  const [yesBook, noBook] = await Promise.all([
    getOrderBook(yesTokenId),
    getOrderBook(noTokenId),
  ]);
  
  const yesBestBid = yesBook.bids[0]?.price;
  const noBestBid = noBook.bids[0]?.price;
  
  if (yesBestBid === undefined || noBestBid === undefined) {
    return false;
  }
  
  // Use best bid (what we'd get selling) for conservative estimate
  const combinedBid = yesBestBid + noBestBid;
  
  return combinedBid < DEFAULT_ARBITRAGE_CONFIG.maxCombinedPrice;
}
```
