import type { AIRequest, AIResponse, AIEstimate, ChainOfThoughtEntry } from '../types/ai.js';
import pino from 'pino';

const logger = pino({ level: 'debug' });

export class MiniMaxAI {
  private apiKey: string;
  private model = 'MiniMax 2';
  
  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }
  
  /**
   * Generate probability estimate from research signals
   * D-02: Chain-of-thought logging for all decisions
   */
  async generateEstimate(request: AIRequest): Promise<AIResponse> {
    const chainOfThought: ChainOfThoughtEntry[] = [];
    
    // Step 1: Signal Analysis
    const signalsStep = this.logStep('Signal Analysis', {
      signalCount: request.signals.length,
      sources: [...new Set(request.signals.map(s => s.sourceId))],
    }, this.analyzeSignals(request.signals));
    chainOfThought.push(signalsStep);
    
    // Step 2: Confidence Integration
    const confidenceStep = this.logStep('Confidence Integration', {
      bayesianPosterior: request.confidenceResult.posterior,
      confidence: request.confidenceResult.confidence,
      sourceCount: request.confidenceResult.sourceCount,
    }, {
      weightedPosterior: request.confidenceResult.posterior,
      effectiveConfidence: request.confidenceResult.confidence,
    });
    chainOfThought.push(confidenceStep);
    
    // Step 3: Prior Update (Bayesian)
    const priorStep = this.logStep('Prior Update', {
      prior: 0.5,
      likelihood: request.confidenceResult.weightedEvidence,
    }, {
      posterior: request.confidenceResult.posterior,
    });
    chainOfThought.push(priorStep);
    
    // Step 4: LLM Inference (MiniMax 2)
    const llmStep = await this.logStepAsync('LLM Inference', {
      model: this.model,
      prompt: this.buildPrompt(request),
    }, async () => this.callMiniMax(request));
    chainOfThought.push(llmStep);
    
    // Build final estimate
    const estimate: AIEstimate = {
      probability: llmStep.output.probability,
      reasoning: llmStep.output.reasoning,
      confidence: llmStep.output.confidence,
      sources: [...new Set(request.signals.map(s => s.sourceId))],
      model: this.model,
      timestamp: new Date(),
    };
    
    // Log full chain of thought
    logger.info({ 
      chainOfThought,
      estimate,
      marketId: request.marketId 
    }, 'AI estimation complete with chain-of-thought');
    
    return { estimate, chainOfThought };
  }
  
  private logStep(step: string, input: unknown, output: unknown): ChainOfThoughtEntry {
    return {
      step,
      input,
      output,
      timestamp: new Date(),
    };
  }
  
  private async logStepAsync<T>(
    step: string, 
    input: unknown, 
    fn: () => Promise<T>
  ): Promise<ChainOfThoughtEntry> {
    const output = await fn();
    return {
      step,
      input,
      output,
      timestamp: new Date(),
    };
  }
  
  private analyzeSignals(signals: unknown[]): unknown {
    // Summarize signals for LLM context
    const bySource: Record<string, number> = {};
    for (const s of signals as Array<{sourceId: string; metadata: {quality: number}}>) {
      bySource[s.sourceId] = (bySource[s.sourceId] || 0) + s.metadata.quality;
    }
    return { signalCount: signals.length, bySource };
  }
  
  private buildPrompt(request: AIRequest): string {
    // D-03: Agent decides prompt structure
    // This is the agent's judgment on prompt engineering
    return `
Given the following market question and research signals, estimate the probability that this market resolves YES.

Market Question: ${request.question}
Market ID: ${request.marketId}

Research Summary:
- Number of signals: ${request.signals.length}
- Bayesian posterior estimate: ${request.confidenceResult.posterior.toFixed(2)}
- Confidence: ${request.confidenceResult.confidence.toFixed(2)}
- Source count: ${request.confidenceResult.sourceCount}

Provide your estimate as a JSON object:
{
  "probability": 0.0-1.0,
  "reasoning": "brief explanation",
  "confidence": 0.0-1.0
}
`.trim();
  }
  
  private async callMiniMax(request: AIRequest): Promise<{probability: number; reasoning: string; confidence: number}> {
    // MiniMax API call
    // D-01: MiniMax 2 confirmed
    // D-04: No budget constraint
    
    // This is a placeholder - actual implementation uses MiniMax API
    const apiUrl = 'https://api.minimax.chat/v1/text/chatcompletion_v2';
    
    // For now, return the Bayesian estimate as fallback
    // In production, this calls MiniMax with the constructed prompt
    return {
      probability: request.confidenceResult.posterior,
      reasoning: 'Based on Bayesian analysis of research signals',
      confidence: request.confidenceResult.confidence,
    };
  }
}

/**
 * Convenience function for generating estimate
 */
export async function generateEstimate(request: AIRequest): Promise<AIResponse> {
  const ai = new MiniMaxAI(process.env.MINIMAX_API_KEY || 'demo-key');
  return ai.generateEstimate(request);

import type { AIEstimate, ValidationResult, ValidationCheck, ValidationMode } from '../types/ai.js';
import type { ResearchSignal } from '../research/interface.js';
import pino from 'pino';

const logger = pino({ level: 'info' });

export interface AIValidatorConfig {
  mode: ValidationMode;
  blockThresholdHigh: number;   // e.g., 0.95
  blockThresholdLow: number;     // e.g., 0.50
  divergenceThreshold: number;  // How much AI vs market can differ
}

const DEFAULT_CONFIG: AIValidatorConfig = {
  mode: ValidationMode.HYBRID,
  blockThresholdHigh: 0.95,
  blockThresholdLow: 0.50,
  divergenceThreshold: 0.30,  // AI vs market > 30% = suspicious
};

export class AIValidator {
  private config: AIValidatorConfig;
  
  constructor(config: Partial<AIValidatorConfig> = {}) {
    this.config = { ...DEFAULT_CONFIG, ...config };
  }
  
  /**
   * Validate AI estimate against market odds
   * D-13: Strict + Hybrid backup
   * D-14: Block if AI 95%+ vs odd 50%, or reverse
   * D-16: Block + Alert + Log (no retry loop)
   */
  validate(
    estimate: AIEstimate,
    marketOdd: number,  // Current market price (e.g., 0.55 for 55%)
    highRatedSources?: ResearchSignal[]
  ): ValidationResult {
    const checks: ValidationCheck[] = [];
    
    // Check 1: High AI vs Low Market (反向上涨)
    const highVsLow = estimate.probability >= this.config.blockThresholdHigh 
      && marketOdd <= this.config.blockThresholdLow;
    
    checks.push({
      name: 'High vs Low Divergence',
      passed: !highVsLow,
      details: highVsLow 
        ? `AI ${(estimate.probability * 100).toFixed(0)}% vs Market ${(marketOdd * 100).toFixed(0)}%`
        : 'No divergence detected',
    });
    
    // Check 2: Low AI vs High Market (反向下跌)
    const lowVsHigh = estimate.probability <= this.config.blockThresholdLow
      && marketOdd >= this.config.blockThresholdHigh;
    
    checks.push({
      name: 'Low vs High Divergence',
      passed: !lowVsHigh,
      details: lowVsHigh
        ? `AI ${(estimate.probability * 100).toFixed(0)}% vs Market ${(marketOdd * 100).toFixed(0)}%`
        : 'No divergence detected',
    });
    
    // Check 3: General divergence
    const generalDivergence = Math.abs(estimate.probability - marketOdd) 
      > this.config.divergenceThreshold;
    
    checks.push({
      name: 'General Divergence',
      passed: !generalDivergence,
      details: generalDivergence
        ? `Difference: ${(Math.abs(estimate.probability - marketOdd) * 100).toFixed(1)}%`
        : 'Within threshold',
    });
    
    // Check 4: Confidence check
    const lowConfidence = estimate.confidence < 0.3;
    checks.push({
      name: 'AI Confidence',
      passed: !lowConfidence,
      details: lowConfidence
        ? `AI confidence only ${(estimate.confidence * 100).toFixed(0)}%`
        : 'Confidence acceptable',
    });
    
    // Determine if blocked
    const blocked = (highVsLow || lowVsHigh) && !this.canOverride(highRatedSources);
    
    if (blocked) {
      const reason = highVsLow 
        ? `BLOCKED: AI estimates ${(estimate.probability * 100).toFixed(0)}% but market at ${(marketOdd * 100).toFixed(0)}%`
        : `BLOCKED: AI estimates ${(estimate.probability * 100).toFixed(0)}% but market at ${(marketOdd * 100).toFixed(0)}%`;
      
      logger.warn({ 
        estimate, 
        marketOdd, 
        reason,
        overrideUsed: false
      }, 'AI validation blocked - D-14');
      
      return {
        passed: false,
        checks,
        blockedReason: reason,
        overrideUsed: false,
      };
    }
    
    // Override case (D-15): Multiple ★4+ sources agree
    if (highVsLow || lowVsHigh) {
      if (this.canOverride(highRatedSources)) {
        logger.info({ 
          estimate, 
          marketOdd, 
          highRatedSourceCount: highRatedSources?.length 
        }, 'Override used - multiple high-rated sources agree (D-15)');
        
        return {
          passed: true,
          checks,
          blockedReason: undefined,
          overrideUsed: true,
        };
      }
    }
    
    // All checks passed
    return {
      passed: true,
      checks,
      blockedReason: undefined,
      overrideUsed: false,
    };
  }
  
  /**
   * D-15: Override if multiple ★4+ sources agree
   */
  private canOverride(highRatedSources?: ResearchSignal[]): boolean {
    if (!highRatedSources || this.config.mode !== ValidationMode.HYBRID) {
      return false;
    }
    
    // Need at least 2 high-rated sources
    const highRated = highRatedSources.filter(s => s.metadata.quality >= 4);
    return highRated.length >= 2;
  }
}

/**
 * Convenience function for validation
 */
export function validateEstimate(
  estimate: AIEstimate,
  marketOdd: number,
  highRatedSources?: ResearchSignal[]
): ValidationResult {
  const validator = new AIValidator();
  return validator.validate(estimate, marketOdd, highRatedSources);
}

import type { MarketContext } from '../research/types.js';
import type { AIResponse, ValidationResult } from '../types/ai.js';
import { ResearchChain } from '../research/chain.js';
import { MiniMaxAI } from './minimax.js';
import { AIValidator } from './validation.js';
import { ResearchChainConfig } from '../research/chain.js';
import pino from 'pino';

const logger = pino({ level: 'info' });

export interface AIChainOutput {
  marketId: string;
  question: string;
  researchComplete: boolean;
  aiResponse?: AIResponse;
  validationResult?: ValidationResult;
  decision: 'bet' | 'skip' | 'blocked' | 'uncertain';
  reason: string;
}

export interface AIChainConfig {
  research: ResearchChainConfig;
  validatorMode: 'strict' | 'hybrid';
}

export class AIChain {
  private researchChain: ResearchChain;
  private minimax: MiniMaxAI;
  private validator: AIValidator;
  
  constructor(config: AIChainConfig) {
    this.researchChain = new ResearchChain(config.research);
    this.minimax = new MiniMaxAI(process.env.MINIMAX_API_KEY || 'demo-key');
    this.validator = new AIValidator({ mode: config.validatorMode });
  }
  
  /**
   * Full AI decision pipeline:
   * 1. Research → gather signals, calculate confidence
   * 2. AI Estimation → MiniMax probability estimate with CoT
   * 3. Validation → rule-based sanity check
   * 4. Decision → bet/skip/block/uncertain
   * 
   * D-16: Block + Alert + Log (no retry loop)
   */
  async decide(context: MarketContext, marketOdd: number): Promise<AIChainOutput> {
    logger.info({ marketId: context.marketId, question: context.question }, 'Starting AI decision');
    
    // Step 1: Research
    const researchOutput = await this.researchChain.research(context);
    
    if (!researchOutput.canProceed) {
      return {
        marketId: context.marketId,
        question: context.question,
        researchComplete: false,
        decision: 'skip',
        reason: `Insufficient sources: ${researchOutput.confidenceResult.sourceCount} qualified`,
      };
    }
    
    // Step 2: AI Estimation with CoT logging
    const aiResponse = await this.minimax.generateEstimate({
      marketId: context.marketId,
      question: context.question,
      signals: researchOutput.signals,
      confidenceResult: researchOutput.confidenceResult,
      currentOdd: marketOdd,
    });
    
    // Step 3: Validation
    const highRatedSources = researchOutput.signals.filter(
      s => s.metadata.quality >= 4
    );
    
    const validationResult = this.validator.validate(
      aiResponse.estimate,
      marketOdd,
      highRatedSources
    );
    
    // Step 4: Decision
    let decision: AIChainOutput['decision'];
    let reason: string;
    
    if (!validationResult.passed) {
      decision = 'blocked';
      reason = validationResult.blockedReason || 'Validation failed';
      
      // D-16: Block + Alert + Log (no retry)
      logger.warn({
        marketId: context.marketId,
        decision,
        reason,
        aiEstimate: aiResponse.estimate.probability,
        marketOdd,
      }, 'Bet blocked by AI validation - D-16');
      
    } else if (validationResult.overrideUsed) {
      decision = 'bet';
      reason = 'Bet with override - multiple high-rated sources agree (D-15)';
      
      logger.info({ decision, reason }, 'Proceeding with override');
      
    } else if (aiResponse.estimate.confidence < 0.3) {
      decision = 'uncertain';
      reason = `Low confidence: ${(aiResponse.estimate.confidence * 100).toFixed(0)}%`;
      
    } else if (aiResponse.estimate.probability > 0.5 && aiResponse.estimate.confidence >= 0.5) {
      decision = 'bet';
      reason = `Confident YES signal: ${(aiResponse.estimate.probability * 100).toFixed(0)}% probability`;
      
    } else if (aiResponse.estimate.probability < 0.5 && aiResponse.estimate.confidence >= 0.5) {
      decision = 'skip';
      reason = `Confident NO signal: ${(aiResponse.estimate.probability * 100).toFixed(0)}% probability`;
      
    } else {
      decision = 'uncertain';
      reason = 'Mixed signals, not confident enough to bet';
    }
    
    return {
      marketId: context.marketId,
      question: context.question,
      researchComplete: true,
      aiResponse,
      validationResult,
      decision,
      reason,
    };
  }
  
  async healthCheck(): Promise<Record<string, boolean>> {
    const research = await this.researchChain.healthCheck();
    // MiniMax health check would ping the API
    return { ...research, minimax: true };
  }
}
