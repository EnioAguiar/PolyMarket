```typescript
import { SourceCategory } from '../types/source.js';

export interface ResearchSignal {
  sourceId: string;
  sourceName: string;
  category: SourceCategory;
  timestamp: Date;
  data: unknown;  // Raw signal data from source
  metadata: {
    confidence: number;  // 0-1 based on source rating
    recency: number;     // Age in seconds
    quality: number;      // Source star rating normalized
  };
}

export interface ResearchSource {
  readonly id: string;
  readonly name: string;
  readonly category: SourceCategory;
  readonly feedType: 'rest' | 'websocket' | 'scraping';
  
  /**
   * Fetch research signals for a given market/question
   * @param marketId - Polymarket condition ID
   * @param question - The market question text
   * @param timeHorizon - Market resolution time (5min, 6h, 24h)
   */
  fetch(marketId: string, question: string, timeHorizon: string): Promise<ResearchSignal[]>;
  
  /**
   * Check if source is healthy and available
   */
  healthCheck(): Promise<boolean>;
}
```

```typescript
export class GoogleNewsAdapter implements ResearchSource {
  readonly id = 'google-news';
  readonly name = 'Google News API';
  readonly category = SourceCategory.NEWS;
  readonly feedType = 'rest' as const;
  
  private apiKey: string;
  
  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }
  
  async fetch(marketId: string, question: string, timeHorizon: string): Promise<ResearchSignal[]> {
    const signals: ResearchSignal[] = [];
    
    // D-25: News time window matches market horizon
    const timeWindow = this.getTimeWindow(timeHorizon);
    
    // 1. Google Search API for related queries
    const searchResults = await this.search(question, timeWindow);
    
    // 2. Google News API for recent news
    const newsResults = await this.news(question, timeWindow);
    
    // 3. Weight by recency (D-20: 2/3 recency + 1/3 age weight)
    const combined = [...searchResults, ...newsResults];
    
    for (const item of combined) {
      const age = (Date.now() - item.publishedAt.getTime()) / 1000;
      
      signals.push({
        sourceId: this.id,
        sourceName: this.name,
        category: this.category,
        timestamp: item.publishedAt,
        data: item,
        metadata: {
          confidence: this.calculateConfidence(item, age),
          recency: age,
          quality: 3,  // ★3 for news (lower than exchange data)
        },
      });
    }
    
    return signals;
  }
  
  private getTimeWindow(timeHorizon: string): string {
    // D-25: Match news time window to market horizon
    switch (timeHorizon) {
```

```typescript
import type { ResearchSource, ResearchSignal } from './interface.js';
import { SourceCategory } from '../types/source.js';
import type { MarketContext } from './types.js';

export class ResearchAggregator {
  private sources: Map<SourceCategory, ResearchSource[]>;
  
  constructor() {
    this.sources = new Map();
  }
  
  registerSource(source: ResearchSource): void {
    const existing = this.sources.get(source.category) || [];
    existing.push(source);
    this.sources.set(source.category, existing);
  }
  
  async aggregate(context: MarketContext): Promise<ResearchSignal[]> {
    const allSignals: ResearchSignal[] = [];
    
    // Get relevant sources for market category
    const relevantSources = this.sources.get(context.category) || [];
    
    // Fetch from each source in parallel
    const results = await Promise.allSettled(
      relevantSources.map(source => 
        source.fetch(context.marketId, context.question, context.timeHorizon)
      )
    );
    
    // Collect successful results
    for (const result of results) {
      if (result.status === 'fulfilled') {
        allSignals.push(...result.value);
      }
      // Log failures but continue
    }
    
    return allSignals;
  }
  
  // D-18: Research mix enforcement
  async aggregateWithMix(
    context: MarketContext,
    binanceWeight = 0.67,
    googleWeight = 0.33
  ): Promise<ResearchSignal[]> {
    const allSignals = await this.aggregate(context);
    
    // Separate by category
    const binanceSignals = allSignals.filter(s => s.sourceId === 'binance');
    const googleSignals = allSignals.filter(s => s.sourceId === 'google-news');
    
    // Apply mix ratio (D-18)
    const targetBinance = Math.floor(allSignals.length * binanceWeight);
    const targetGoogle = Math.floor(allSignals.length * googleWeight);
    
    // If we have more Binance signals than target, trim
    // If fewer, that's OK per D-12 (relaxed policy with high-rated sources)
    const trimmed = [
      ...binanceSignals.slice(0, targetBinance),
      ...googleSignals.slice(0, targetGoogle),
    ];
    
    return trimmed;
  }
  
  async healthCheck(): Promise<Record<string, boolean>> {
    const status: Record<string, boolean> = {};
    
    for (const [category, sources] of this.sources.entries()) {
      for (const source of sources) {
        status[source.id] = await source.healthCheck();
      }
    }
    
    return status;
  }
}
```

Key points:
- Aggregates signals from multiple sources
- Respects research mix ratio (D-18: 2/3 Binance, 1/3 Google)
- Handles source failures gracefully
- Health check for all registered sources
