| updated_at | timestamp | Last update |

**Table 2: source_feeds**
| Column | Type | Description |
|--------|------|-------------|
| id | text PK | Feed identifier |
| source_id | text FK | References source_ratings |
| feed_url | text | Actual endpoint/URL |
| feed_type | text | rest, websocket, scraping |
| enabled | boolean | Feed active state |
| last_fetch | timestamp | Last successful fetch |

**Table 3: research_results**
| Column | Type | Description |
|--------|------|-------------|
| id | text PK | UUID |
| market_id | text | Polymarket condition ID |
| source_id | text FK | Source that provided data |
| signal | text | JSON signal data |
| confidence | real | 0-1 confidence score |
| fetched_at | timestamp | When data was fetched |
| processed | boolean | Has been fed to AI |

```typescript
// src/types/source.ts

export enum SourceCategory {
  CRYPTO = 'crypto',
  NEWS = 'news',
  FINANCIAL = 'financial',
  SPORTS = 'sports',
}

export enum FeedType {
  REST = 'rest',
  WEBSOCKET = 'websocket',
  SCRAPING = 'scraping',
}

export interface SourceRating {
  id: string;
  name: string;
  category: SourceCategory;
  rating: 1 | 2 | 3 | 4 | 5;  // ★1-5
  apiEndpoint: string;
  config: Record<string, unknown>;
  createdAt: Date;
  updatedAt: Date;
}

export interface FeedConfig {
  id: string;
  sourceId: string;
  feedUrl: string;
  feedType: FeedType;
  enabled: boolean;
  lastFetch: Date | null;
}

export interface ResearchResult {
  id: string;
  marketId: string;
  sourceId: string;
  signal: unknown;  // JSON data from source
  confidence: number;  // 0-1
  fetchedAt: Date;
  processed: boolean;
}

// Star rating thresholds
export const MIN_RATING = 3;  // D-12: Minimum ★3 enforced
export const MIN_SOURCES = 10;  // D-27: 10 source minimum
export const HIGH_RATING = 4;  // D-12: ★4+ can override relaxed policy
```

Export all types. Location: `src/types/source.ts`
</action>

  </verify>
