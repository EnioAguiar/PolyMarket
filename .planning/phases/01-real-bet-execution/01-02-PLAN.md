```typescript
export async function placeMarketOrder(
  orderbook: OrderBook,
  side: 'BUY' | 'SELL',
  size: number,  // dollar amount to spend
  bankroll: number,  // total bankroll in USDC
  config: { maxSlippagePct?: number } = {}
): Promise<OrderResult> {
  const logger = getLogger();
  
  // D-15: 5% fixed bankroll position sizing
  const maxPosition = bankroll * 0.05;
  
  if (size > maxPosition) {
    logger.warn({ size, maxPosition, bankroll }, 'Position size exceeds 5% bankroll limit');
    return {
      success: false,
      orderType: OrderType.MARKET,
      reason: `Position size $${size} exceeds 5% bankroll limit $${maxPosition.toFixed(2)}`,
    };
  }
  
  // Check slippage before execution
  const executionPrice = side === 'BUY' ? orderbook.asks[0]?.price : orderbook.bids[0]?.price;
  const expectedPrice = getMidPrice(orderbook);
  
  if (expectedPrice && executionPrice) {
    const slippageResult = checkSlippage(
      { expectedPrice, executionPrice },
      config.maxSlippagePct ?? 0.10  // D-09: 10% default
    );
    
    if (!slippageResult.allowed) {
      logger.warn({ slippage: slippageResult }, 'Slippage exceeded - aborting');
      return {
        success: false,
        orderType: OrderType.MARKET,
        reason: `Slippage ${(slippageResult.slippagePct * 100).toFixed(1)}% exceeds ${(slippageResult.threshold * 100).toFixed(0)}%`,
      };
    }
  }
  
  // ... rest of execution
}
```

Note: This is the local execution module. The actual API call happens in src/api/clob.ts. The execution module validates before calling the API.
