## What Changes

**Before (cron-based):**
- Railway cron triggers every 5 minutes
- src/index.ts starts HTTP server → runs runBotCycle() → exits

**After (event-driven):**
- Railway runs 24/7 (no cron)
- src/index.ts starts HTTP server → connects WebSocket → runs forever
- On WebSocket events, trigger evaluateMarket for qualifying markets

async function main(): Promise<void> {
  const config = loadConfig();
  initLogger(config);
  const logger = getLogger();
  
  logger.info({ 
    dryRun: config.dryRun,
    msg: 'Polymarket Bot starting (event-driven mode)',
  });
  
  try {
    await startServer();
    
    // Initialize safety module
    const initialState: SafetyState = {
      dailyLoss: 0,
      totalDrawdown: 0,
      isKillSwitchActive: false,
    };
    const initialBankroll = 1000;
    const safetyModule = new SafetyModule(config, initialState, initialBankroll);
    
    // Create CLOB client if not dry run
    let clobClient = null;
    if (!config.dryRun) {
      clobClient = createClobClient(config);
    }
    
    // Create WebSocket client
    const wsClient = new PolymarketWsClient({
      wsUrl: 'wss://ws-subscriptions-clob.polymarket.com/ws/market',
      heartbeatInterval: 10000,
      reconnectBaseDelay: 1000,
      reconnectMaxDelay: 30000,
    });
    
    // Set up event handlers
    const eventHandlers: WsEventHandlers = {
      onNewMarket: (event) => {
        logger.info({ 
          marketId: event.market, 
          question: event.question?.substring(0, 60),
          assets: event.assets_ids.length,
        }, 'New market event received');
        
        // Evaluate market through existing pipeline
        evaluateMarketForWebSocket(event, safetyModule, clobClient, config, logger);
      },
      onBestBidAsk: (event) => {
        logger.debug({ 
          assetId: event.asset_id,
          bestBid: event.best_bid,
          bestAsk: event.best_ask,
        }, 'Best bid/ask update');
        
        // Update odds cache for monitored markets
        updateOddsFromWs(event.asset_id, parseFloat(event.best_bid), parseFloat(event.best_ask));
      },
      onPriceChange: (event) => {
        logger.debug({ 
          market: event.market,
          changes: event.price_changes.length,
        }, 'Price change event');
      },
      onMarketResolved: (event) => {
        logger.info({ 
          marketId: event.market,
          winningOutcome: event.winning_outcome,
        }, 'Market resolved');
        
        // Close position / record outcome
        handleMarketResolved(event.market, event.winning_outcome, logger);
      },
    };
    
    // Wire up client event handling
    wsClient.onEvent((event) => handleWsEvent(event, eventHandlers));
    
    wsClient.onConnect(() => {
      logger.info({ msg: 'WebSocket connected to Polymarket' });
      isHealthy = true;
    });
    
    wsClient.onDisconnect(() => {
      logger.warn({ msg: 'WebSocket disconnected, attempting reconnect' });
      isHealthy = false;
    });
    
    wsClient.onError((error) => {
      logger.error({ error, msg: 'WebSocket error' });
    });
    
    // Connect with initial asset list (empty = all markets via new_market events)
    const initialAssets = config.monitoredAssets || [];
    wsClient.connect(initialAssets);
    
    // Keep process alive
    logger.info({ msg: 'Bot running, press Ctrl+C to stop' });
    
  } catch (error) {
    logger.error({ error, msg: 'Fatal error in bot startup' });
    isHealthy = false;
    process.exit(1);
  }
}
```

**Graceful shutdown:**

// Handle SIGINT/SIGTERM for graceful shutdown
process.on('SIGINT', () => {
  logger.info({ msg: 'SIGINT received, shutting down gracefully' });
  wsClient?.disconnect();
  process.exit(0);
});

process.on('SIGTERM', () => {

  wsClient?.disconnect();
  process.exit(0);
});
```

**Keep-alive:** The process stays running because wsClient.connect() doesn't return (it's async but the connection is persistent). No setInterval needed.

// Evaluate a market from WebSocket new_market event
export async function evaluateMarketForWebSocket(
  event: WsMarketEvent,
  safetyModule: SafetyModule,
  clobClient: any,
  config: Config,
  logger: Logger
): Promise<void> {
  // Filter by time horizon (5min - 24h)
  const resolveTime = new Date(event.endDate || event.game_start_time);
  const now = Date.now();
  const timeUntilResolve = resolveTime.getTime() - now;
  const minMs = 5 * 60 * 1000;
  const maxMs = 24 * 60 * 60 * 1000;
  
  if (timeUntilResolve < minMs || timeUntilResolve > maxMs) {
    logger.debug({ marketId: event.market, timeUntilResolve }, 'Market outside time horizon, skipping');
    return;
  }
  
  // Get YES token ID (first asset)
  const yesTokenId = event.assets_ids[0];
  if (!yesTokenId) return;
  
  // Get orderbook via CLOB
  const orderbook = await getOrderBook(yesTokenId);
  if (!hasLiquidity(orderbook, 1)) {
    logger.debug({ marketId: event.market }, 'Insufficient liquidity');
    return;
  }
  
  const odds = getMidPrice(orderbook);
  if (odds === null) return;
  
  const bankroll = 1000;
  const maxPosition = safetyModule.getMaxPositionSizeForOdds(odds);
  
  const safetyResult = safetyModule.checkBet({ odds, positionSize: maxPosition, bankroll });
  if (!safetyResult.passed) {
    logger.info({ marketId: event.market, reason: safetyResult.message }, 'Safety check failed');
    return;
  }
  
  // Log bet decision (Phase 1 = monitoring only)
  logBetDecision({
    marketId: event.market,
    odds,
    positionSize: maxPosition,
    side: 'BUY',
    dryRun: config.dryRun,
    safetyCheck: 'passed',
    reason: 'Event-driven evaluation from WebSocket',
    timestamp: new Date().toISOString(),
  });
}

// Update cached odds from best_bid_ask events
const oddsCache = new Map<string, { bid: number; ask: number; timestamp: number }>();

export function updateOddsFromWs(assetId: string, bid: number, ask: number): void {
  oddsCache.set(assetId, { bid, ask, timestamp: Date.now() });
}

export function getCachedOdds(assetId: string): { bid: number; ask: number } | null {
  const cached = oddsCache.get(assetId);
  if (!cached) return null;
  
  // Cache expires after 60 seconds
  if (Date.now() - cached.timestamp > 60000) {
    oddsCache.delete(assetId);
    return null;
  }
  
  return { bid: cached.bid, ask: cached.ask };
}
