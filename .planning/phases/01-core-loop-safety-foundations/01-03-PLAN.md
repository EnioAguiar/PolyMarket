## Phase 1 Success Criteria

1. Bot connects to Polymarket API and fetches market list filtered by category and time (5min-24h)
2. Bot fetches odds and orderbook depth for selected markets
3. Bot operates in dry-run mode (logs decisions without executing trades)
4. Max position size enforced (5-10% per market, configurable)
5. Daily loss limit enforced (stops trading when threshold exceeded)
6. Drawdown kill switch halts all trading on extreme drawdown)

import { loadConfig, isDryRun } from './config/index.js';
import { initLogger, getLogger, logBetDecision } from './logging/index.js';
import { fetchMarkets, filterByCategory, filterByTimeHorizon, getYesTokenId } from './api/polymarket.js';
import { getOrderBook, getMidPrice, createClobClient, hasLiquidity } from './api/clob.js';
import { SafetyModule } from './safety/index.js';
import type { Market, SafetyState } from './types/index.js';

/**
 * Main bot loop
 * Called by Railway cron every 5 minutes (per D-01)
 * Cycles: monitor → research → analyze → decide → execute (dry-run logged)
 */
export async function runBotCycle(): Promise<void> {
  const logger = getLogger();
  const config = loadConfig();
  
  logger.info({ dryRun: config.dryRun }, 'Bot cycle starting');
  
  // Initialize CLOB client if not in dry-run mode
  let clobClient = null;
  if (!config.dryRun) {
    clobClient = createClobClient(config);
  }
  
  // Initialize safety module with initial state
  const initialState: SafetyState = {
    dailyLoss: 0,
    totalDrawdown: 0,
    isKillSwitchActive: false,
  };
  const initialBankroll = 1000; // TODO: Load from persistent storage (Phase 2/3)
  const safetyModule = new SafetyModule(config, initialState, initialBankroll);
  
  // Check kill switch first
  if (safetyModule.isKillSwitchActive()) {
    logger.warn({ msg: 'KILL SWITCH ACTIVE - bot halted' });
    return;
  }
  
  // Step 1: Monitor markets
  logger.info({ step: 'monitor' }, 'Fetching markets from Polymarket');
  const allMarkets = await fetchMarkets({ limit: 50 });
  
  // Filter by category (configurable - default to crypto for initial implementation)
  const category = process.env.MARKET_CATEGORY || 'crypto';
  const filteredByCategory = filterByCategory(allMarkets, category);
  
  // Filter by time horizon (5min to 24h per success criteria)
  const filteredByTime = filterByTimeHorizon(filteredByCategory, 5, 24);
  
  logger.info({ 
    totalMarkets: allMarkets.length,
    category,
    filteredByTime: filteredByTime.length,
  }, 'Markets filtered');
  
  if (filteredByTime.length === 0) {
    logger.info({ msg: 'No markets match criteria, skipping cycle' });
    return;
  }
  
  // Step 2 & 3: Research and analyze (simplified for Phase 1)
  // Phase 2 will add full research pipeline with star ratings
  logger.info({ step: 'analyze' }, 'Analyzing markets');
  
  for (const market of filteredByTime) {
    const yesTokenId = getYesTokenId(market);
    if (!yesTokenId) continue;
    
    // Skip if safety kill switch activated during loop
    if (safetyModule.isKillSwitchActive()) {
      logger.warn({ marketId: market.id, msg: 'Kill switch activated mid-cycle, halting' });
      break;
    }
    
    try {
      // Step 4: Decide and execute
      const decision = await evaluateMarket(market, safetyModule, config);
      
      // Log the decision (dry-run or real)
      logBetDecision({
        marketId: market.id,
        odds: decision.odds,
        positionSize: decision.positionSize,
        dryRun: config.dryRun,
        action: decision.action,
        safetyCheck: decision.safetyCheck,
        reason: decision.reason,
      });
      
      // In Phase 3+ this would execute real trades
      // For Phase 1, dry-run logs the decision
    } catch (error) {
      logger.error({ marketId: market.id, error }, 'Error processing market');
    }
  }
  
  logger.info({ step: 'complete' }, 'Bot cycle complete');
}

/**
 * Evaluate a single market and decide whether to place a bet
 */
async function evaluateMarket(
  market: Market,
  safetyModule: SafetyModule,
  config: ReturnType<typeof loadConfig>
): Promise<{
  odds: number;
  positionSize: number;
  action: 'monitor' | 'analyze' | 'bet' | 'skip';
  safetyCheck: string;
  reason: string;
}> {
  const logger = getLogger();
  const yesTokenId = getYesTokenId(market);
  
  if (!yesTokenId) {
    return {
      odds: 0,
      positionSize: 0,
      action: 'skip',
      safetyCheck: 'none',
      reason: 'No YES token ID',
    };
  }
  
  // Get orderbook and odds
  let orderbook;
  try {
    orderbook = await getOrderBook(yesTokenId, 5);
  } catch (error) {
    return {
      odds: 0,
      positionSize: 0,
      action: 'skip',
      safetyCheck: 'none',
      reason: `Failed to fetch orderbook: ${error}`,
    };
  }
  
  // Check liquidity
  if (!hasLiquidity(orderbook, 1)) {
    return {
      odds: 0,
      positionSize: 0,
      action: 'skip',
      safetyCheck: 'none',
      reason: 'Insufficient liquidity',
    };
  }
  
  const odds = getMidPrice(orderbook);
  if (odds === null) {
    return {
      odds: 0,
      positionSize: 0,
      action: 'skip',
      safetyCheck: 'none',
      reason: 'No mid-price available',
    };
  }
  
  // Calculate max position size based on odds
  const bankroll = 1000; // TODO: Load from persistent state
  const maxPosition = safetyModule.getMaxPositionSizeForOdds(odds);
  
  // Safety check
  const safetyResult = safetyModule.checkBet({
    odds,
    positionSize: maxPosition,
    bankroll,
  });
  
  if (!safetyResult.passed) {
    return {
      odds,
      positionSize: maxPosition,
      action: 'skip',
      safetyCheck: safetyResult.checkType,
      reason: safetyResult.message || 'Safety check failed',
    };
  }
  
  // In Phase 1, we just monitor/log - no actual betting yet
  // Phase 2 adds source intelligence
  // Phase 3 adds real execution with Kelly criterion
  return {
    odds,
    positionSize: maxPosition,
    action: 'monitor', // Just monitoring in Phase 1
    safetyCheck: 'passed',
    reason: 'Phase 1 - monitoring only, no execution',
  };
}

```typescript
import http from 'http';
import { loadConfig } from './config/index.js';
import { initLogger, getLogger } from './logging/index.js';
import { runBotCycle } from './main.js';

// Health check state
let isHealthy = false;

/**
 * Health check endpoint (DEPL-02)
 * Railway pings this to verify service is alive
 */
function healthCheck(req: http.IncomingMessage, res: http.ServerResponse): void {
  if (req.url === '/health' && req.method === 'GET') {
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify({ 
      status: isHealthy ? 'healthy' : 'initializing',
      timestamp: new Date().toISOString(),
      service: 'polymarket-bot',
    }));
    return;
  }
  
  // 404 for all other routes
  res.writeHead(404, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify({ error: 'Not found' }));
}

/**
 * Start the HTTP server (for health checks)
 * Railway still triggers via cron, but we need HTTP for health ping
 */
function startServer(): Promise<http.Server> {
  return new Promise((resolve) => {
    const server = http.createServer(healthCheck);
    const port = process.env.PORT || 3000;
    
    server.listen(port, () => {
      getLogger().info({ port, msg: 'Health check server started' });
      isHealthy = true;
      resolve(server);
    });
  });
}

/**
 * Main entry point
 * Railway cron triggers this every 5 minutes
 * Service starts → runs bot cycle → exits
 */
async function main(): Promise<void> {
  // Initialize logger first
  const config = loadConfig();
  initLogger(config);
  const logger = getLogger();
  
  logger.info({ 
    dryRun: config.dryRun,
    msg: 'Polymarket Bot starting',
  });
  
  try {
    // Start health check server (non-blocking)
    await startServer();
    
    // Run the bot cycle
    await runBotCycle();
    
    logger.info({ msg: 'Bot cycle complete, exiting' });
    isHealthy = false;
    
    // Exit cleanly (Railway cron expects service to exit)
    process.exit(0);
  } catch (error) {
    logger.error({ error, msg: 'Fatal error in bot cycle' });
    isHealthy = false;
    process.exit(1);
  }
}

// Run if executed directly
main().catch((error) => {
  console.error('Unhandled error:', error);
  process.exit(1);
});

// Export for testing
export { main, healthCheck, startServer };
```
