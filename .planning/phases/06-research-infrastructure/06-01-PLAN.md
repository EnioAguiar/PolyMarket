```typescript
      return order[a.priority] - order[b.priority];
    })
    .map(c => c.source);
```

**GoogleNewsAdapter** (`src/research/sources/google.ts`):
- Implements ResearchSource interface
- Category: SourceCategory.NEWS
- Rating: 3 (default)
- Uses Google News API via config.apiKeys.googleNews
- Fetches news articles related to topic
- Returns ResearchSignal with article count, sentiment signals

```typescript
import { BaseResearchAdapter } from './base.js';
import { SourceCategory } from '../../types/source.js';

export class GoogleNewsAdapter extends BaseResearchAdapter {
  id = 'google-news';
  name = 'Google News API';
  category = SourceCategory.NEWS;
  rating = 3 as const;
  
  private apiKey: string;
  
  constructor(apiKey?: string) {
    super();
    this.apiKey = apiKey || process.env.GOOGLE_NEWS_API_KEY || '';
  }
  
  isAvailable(): boolean {
    return this.apiKey.length > 0;
  }
  
  async fetch(topic: string, marketTimeHorizon?: number): Promise<ResearchSignal> {
    if (!this.apiKey) {
      throw new Error('Google News API key not configured');
    }
    
    const url = `https://news.google.com/rss/search?q=${encodeURIComponent(topic)}&hl=en-US&gl=US&ceid=US:en`;
    // Note: Using RSS proxy to avoid CORS issues in Node.js environment
    // For production, use Google News API with proper API key
    const response = await fetch(`https://api.rss2json.com/v1/api.json?rss_url=${encodeURIComponent(url)}&api_key=${this.apiKey}`);
    
    if (!response.ok) {
      throw new Error(`Google News fetch failed: ${response.statusText}`);
    }
    
    const data = await response.json() as { status: string; items: Array<{title: string; description: string; pubDate: string}> };
    
    if (data.status !== 'ok' || !data.items) {
      throw new Error('Google News API returned invalid response');
    }
    
    const articles = data.items || [];
    const recency = this.calculateRecency(articles);
    
    return this.createSignal(
      { articleCount: articles.length, headlines: articles.slice(0, 5).map(a => a.title) },
      0.6 + (articles.length > 3 ? 0.2 : 0),
      { apiSource: 'google-news', recency }
    );
  }
  
  private calculateRecency(articles: Array<{pubDate: string}>): number {
    if (articles.length === 0) return 0;
    const latest = new Date(articles[0].pubDate);
    const hoursAgo = (Date.now() - latest.getTime()) / (1000 * 60 * 60);
    return Math.max(0, 1 - (hoursAgo / 24)); // Decay over 24h
  }
}
```

```typescript
import { BaseResearchAdapter } from './base.js';
import { SourceCategory } from '../../types/source.js';

export class CoinGeckoAdapter extends BaseResearchAdapter {
  id = 'coingecko';
  name = 'CoinGecko API';
  category = SourceCategory.CRYPTO;
  rating = 4 as const;
  
  private readonly BASE_URL = 'https://api.coingecko.com/api/v3';
  
  // Symbol mapping from topic to CoinGecko ids
  private symbolToId: Record<string, string> = {
    'bitcoin': 'bitcoin',
    'btc': 'bitcoin',
    'ethereum': 'ethereum',
    'eth': 'ethereum',
    'solana': 'solana',
    'sol': 'solana',
    'dogecoin': 'dogecoin',
    'doge': 'dogecoin',
  };
  
  async fetch(topic: string, marketTimeHorizon?: number): Promise<ResearchSignal> {
    const coinId = this.symbolToId[topic.toLowerCase()] || this.symbolToId[topic.toLowerCase().split(' ')[0]];
    
    if (!coinId) {
      throw new Error(`No CoinGecko mapping for topic: ${topic}`);
    }
    
    const response = await fetch(`${this.BASE_URL}/coins/${coinId}`, {
      headers: { 'Accept': 'application/json' },
    });
    
    if (!response.ok) {
      throw new Error(`CoinGecko fetch failed: ${response.statusText}`);
    }
    
    const data = await response.json() as {
      market_data: {
        current_price: { usd: number };
        price_change_percentage_24h: number;
        market_cap: { usd: number };
        total_volume: { usd: number };
      };
      community_sentiment_score: number;
      developer_sentiment_score: number;
    };
    
    const priceChange = data.market_data?.price_change_percentage_24h || 0;
    const volume = data.market_data?.total_volume?.usd || 0;
    
    // Higher confidence for high volume and significant price movement
    let confidence = 0.7;
    if (volume > 1_000_000) confidence += 0.1;
    if (Math.abs(priceChange) > 1) confidence += 0.1;
    
    return this.createSignal(
      {
        price: data.market_data.current_price.usd,
        priceChange24h: priceChange,
        volume,
      },
      Math.min(confidence, 1.0),
      { apiSource: 'coingecko', coinId }
    );
  }
}
```

```typescript
import { BaseResearchAdapter } from './base.js';
import { SourceCategory } from '../../types/source.js';

export class FootballAdapter extends BaseResearchAdapter {
  id = 'api-football';
  name = 'API-Football';
  category = SourceCategory.SPORTS;
  rating = 3 as const;
  
  private readonly BASE_URL = 'https://api-football-v1.p.rapidapi.com/v3';
  private apiKey: string;
  
  constructor(apiKey?: string) {
    super();
    this.apiKey = apiKey || process.env.API_FOOTBALL_KEY || '';
  }
  
  isAvailable(): boolean {
    return this.apiKey.length > 0;
  }
  
  async fetch(topic: string, marketTimeHorizon?: number): Promise<ResearchSignal> {
    if (!this.apiKey) {
      throw new Error('API-Football key not configured');
    }
    
    // Parse topic to extract team/league info
    // Topic format: "Team vs Team" or "League: Team"
    const teams = topic.split(' vs ').map(t => t.trim());
    const leagueInfo = topic.split(':')[0]?.trim();
    
    // Search for teams
    const searchQuery = encodeURIComponent(teams[0] || topic);
    const searchRes = await fetch(`${this.BASE_URL}/teams?name=${searchQuery}`, {
      headers: {
        'X-RapidAPI-Key': this.apiKey,
        'X-RapidAPI-Host': 'api-football-v1.p.rapidapi.com',
      },
    });
    
    if (!searchRes.ok) {
      throw new Error(`API-Football search failed: ${searchRes.statusText}`);
    }
    
    const searchData = await searchRes.json() as { response: Array<{ team: { id: number; name: string; logo: string } }> };
    const team = searchData.response?.[0]?.team;
    
    if (!team) {
      throw new Error(`No team found for: ${topic}`);
    }
    
    // Get team form (last 5 matches)
    const fixturesRes = await fetch(`${this.BASE_URL}/fixtures?team=${team.id}&last=5`, {
      headers: {
        'X-RapidAPI-Key': this.apiKey,
        'X-RapidAPI-Host': 'api-football-v1.p.rapidapi.com',
      },
    });
    
    let form = 'unknown';
    let winCount = 0;
    
    if (fixturesRes.ok) {
      const fixturesData = await fixturesRes.json() as {
        response: Array<{
          fixture: { status: { short: string } };
          teams: { home: { id: number; name: string; winner: boolean | null }; away: { id: number; name: string; winner: boolean | null } };
          goals: { home: number; away: number };
        }>
      };
      
      const matches = fixturesData.response || [];
      for (const match of matches) {
        const isHome = match.teams.home.id === team.id;
        const teamWon = isHome ? match.teams.home.winner : match.teams.away.winner;
        const teamLost = isHome ? match.teams.away.winner : match.teams.home.winner;
        if (teamWon) winCount++;
      }
      
      // Calculate win rate
      if (matches.length > 0) {
        const winRate = winCount / matches.length;
        form = winRate >= 0.6 ? 'good' : winRate >= 0.4 ? 'medium' : 'poor';
      }
    }
    
    return this.createSignal(
      {
        team: team.name,
        form,
        winCount,
        totalMatches: 5,
        logo: team.logo,
      },
      0.6 + (winCount / 10), // 0.6 to 1.0 based on form
      { apiSource: 'api-football', teamId: team.id }
    );
  }
}
```

- Implements ResearchSource interface

```typescript
import { BaseResearchAdapter } from './base.js';
import { SourceCategory } from '../../types/source.js';

export class Crawl4AIAdapter extends BaseResearchAdapter {
  id = 'crawl4ai';
  name = 'Crawl4AI Web Scraper';
  category = SourceCategory.NEWS;
  rating = 2 as const; // D-03: Lower rating for scraped data
  
  private apiUrl: string;
  
  constructor(apiUrl?: string) {
    super();
    // Default: assume Crawl4AI is running locally
    this.apiUrl = apiUrl || process.env.CRAWL4AI_URL || 'http://localhost:8000';
  }
  
  isAvailable(): boolean {
    // Check if Crawl4AI service is reachable
    // For now, assume available if URL is configured
    return !!process.env.CRAWL4AI_URL;
  }
  
  async fetch(topic: string, marketTimeHorizon?: number): Promise<ResearchSignal> {
    const url = this.topicToUrl(topic);
    
    if (!url) {
      throw new Error(`No URL mapping for topic: ${topic}`);
    }
    
    try {
      const response = await fetch(`${this.apiUrl}/crawl`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          urls: [url],
          max_depth: 1,
          word_count_threshold: 100,
        }),
      });
      
      if (!response.ok) {
        throw new Error(`Crawl4AI crawl failed: ${response.statusText}`);
      }
      
      const data = await response.json() as {
        results: Array<{ url: string; content: string; success: boolean }>;
      };
      
      const result = data.results?.[0];
      if (!result?.success || !result.content) {
        throw new Error('Crawl4AI failed to extract content');
      }
      
      // Extract keywords and sentiment from content
      const content = result.content;
      const wordCount = content.split(/\s+/).length;
      const sentiment = this.analyzeSentiment(content);
      
      return this.createSignal(
        {
          url,
          wordCount,
          sentiment,
          summary: content.substring(0, 500),
        },
        0.4, // D-03: Lower confidence for scraped data
        { scraper: 'crawl4ai' }
      );
    } catch (error) {
      // Crawl4AI is fallback - throw error to indicate unavailability
      throw new Error(`Crawl4AI fetch failed: ${error}`);
    }
  }
  
  private topicToUrl(topic: string): string | null {
    // URL mappings for common topics
    const mappings: Record<string, string> = {
      'bitcoin': 'https://coinDesk.com',
      'ethereum': 'https://ethereum.org',
      'sports': 'https://espn.com',
    };
    
    const lower = topic.toLowerCase();
    for (const [key, url] of Object.entries(mappings)) {
      if (lower.includes(key)) return url;
    }
    
    // Default: use Google Search for the topic
    return `https://www.google.com/search?q=${encodeURIComponent(topic)}`;
  }
  
  private analyzeSentiment(content: string): 'positive' | 'negative' | 'neutral' {
    const lower = content.toLowerCase();
    const positiveWords = ['rise', 'gain', 'surge', 'bull', 'up', 'increase', 'win', 'success', 'positive'];
    const negativeWords = ['fall', 'drop', 'bear', 'down', 'decrease', 'lose', 'loss', 'negative', 'crash'];
    
    let positiveCount = 0;
    let negativeCount = 0;
    
    for (const word of positiveWords) {
      if (lower.includes(word)) positiveCount++;
    }
    for (const word of negativeWords) {
      if (lower.includes(word)) negativeCount++;
    }
    
    if (positiveCount > negativeCount) return 'positive';
    if (negativeCount > positiveCount) return 'negative';
    return 'neutral';
  }
}
```
