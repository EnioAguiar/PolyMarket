```typescript
import { drizzle } from 'drizzle-orm/better-sqlite3';
import Database from 'better-sqlite3';
import { resolve } from 'path';
import { loadConfig } from '../config/index.js';

let dbInstance: ReturnType<typeof drizzle> | null = null;
let sqliteInstance: Database.Database | null = null;

// D-10: SQLite database at /data/polymarket.db
// D-11: Database survives deploys (not wiped on restart)
export function getDb() {
  if (dbInstance) return dbInstance;
  
  const config = loadConfig();
  
  // Use /data path from config, fallback to ./data for local dev
  // D-09: Volume mount path: /data
  const dbPath = config.database?.path || process.env.DATABASE_PATH || '/data/polymarket.db';
  
  // Ensure directory exists
  const dbDir = resolve(dbPath, '..');
  // Note: In Railway volume environment, /data already exists
  // For local development, ensure the directory is created
  if (!dbPath.startsWith('/data') && process.env.NODE_ENV !== 'production') {
    // Local development: create data directory if needed
    import('fs').then(fs => {
      if (!fs.existsSync('./data')) {
        fs.mkdirSync('./data', { recursive: true });
      }
    });
  }
  
  sqliteInstance = new Database(dbPath);
  dbInstance = drizzle(sqliteInstance);
  
  return dbInstance;
}

// Export for direct sqlite access if needed
export function getSqlite(): Database.Database {
  if (!sqliteInstance) {
    getDb(); // Initialize
  }
  return sqliteInstance!;
}

// Database path used for logging/debugging
export function getDatabasePath(): string {
  return process.env.DATABASE_PATH || '/data/polymarket.db';
}
```

```typescript
import Database from 'better-sqlite3';
import { resolve } from 'path';
import { drizzle } from 'drizzle-orm/better-sqlite3';
import { sql } from 'drizzle-orm';

// D-10: SQLite database at /data/polymarket.db
// D-11: Database survives deploys

const DB_PATH = process.env.DATABASE_PATH || '/data/polymarket.db';

async function initDatabase() {
  console.log(`Initializing database at: ${DB_PATH}`);
  
  // Ensure directory exists
  const dbDir = resolve(DB_PATH, '..');
  if (dbDir !== '/data' && dbDir !== '.') {
    // Only for non-volume paths (local dev)
    await import('fs').then(fs => {
      if (!fs.existsSync(dbDir)) {
        fs.mkdirSync(dbDir, { recursive: true });
      }
    });
  }
  
  // Create database
  const db = new Database(DB_PATH);
  
  // Enable foreign keys
  db.pragma('foreign_keys = ON');
  
  // Create tables using raw SQL (Drizzle can also do this)
  db.exec(`
    CREATE TABLE IF NOT EXISTS source_ratings (
      id TEXT PRIMARY KEY,
      name TEXT NOT NULL,
      category TEXT NOT NULL,
      rating INTEGER NOT NULL,
      api_endpoint TEXT NOT NULL,
      config TEXT,
      created_at TEXT NOT NULL,
      updated_at TEXT NOT NULL
    );
    
    CREATE TABLE IF NOT EXISTS source_feeds (
      id TEXT PRIMARY KEY,
      source_id TEXT NOT NULL REFERENCES source_ratings(id),
      feed_url TEXT NOT NULL,
      feed_type TEXT NOT NULL,
      enabled INTEGER NOT NULL DEFAULT 1,
      last_fetch TEXT
    );
    
    CREATE TABLE IF NOT EXISTS research_results (
      id TEXT PRIMARY KEY,
      market_id TEXT NOT NULL,
      source_id TEXT NOT NULL REFERENCES source_ratings(id),
      signal TEXT,
      confidence REAL NOT NULL,
      fetched_at TEXT NOT NULL,
      processed INTEGER NOT NULL DEFAULT 0
    );
    
    CREATE INDEX IF NOT EXISTS idx_research_market ON research_results(market_id);
    CREATE INDEX IF NOT EXISTS idx_research_source ON research_results(source_id);
    CREATE INDEX IF NOT EXISTS idx_source_category ON source_ratings(category);
  `);
  
  console.log('Database tables created successfully');
  
  // Insert default source configurations
  const now = new Date().toISOString();
  
  const insertSource = db.prepare(`
    INSERT OR IGNORE INTO source_ratings (id, name, category, rating, api_endpoint, config, created_at, updated_at)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?)
  `);
  
  // Default sources
  const defaultSources = [
    ['binance', 'Binance WebSocket', 'crypto', 4, 'wss://stream.binance.com:9443/ws', '{}', now, now],
    ['coingecko', 'CoinGecko API', 'crypto', 4, 'https://api.coingecko.com/api/v3', '{}', now, now],
    ['google-news', 'Google News', 'news', 3, 'https://news.google.com/rss', '{}', now, now],
    ['api-football', 'API-Football', 'sports', 3, 'https://api-football-v1.p.rapidapi.com/v3', '{}', now, now],
    ['crawl4ai', 'Crawl4AI Web Scraper', 'news', 2, 'http://localhost:8000', '{}', now, now],
  ];
  
  for (const source of defaultSources) {
    insertSource.run(...source);
  }
  
  console.log('Default sources inserted');
  
  db.close();
  console.log('Database initialization complete');
}

initDatabase().catch(console.error);
```
